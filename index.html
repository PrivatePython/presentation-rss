5. Responsive Design

import { Box } from '@chakra-ui/react';

	return (
		<Box
				width={["100%", "50%", "25%"]}

				>
			Responsive Box
		</Box>
	);

6. CSS-in-JS

import React from 'react';
import { Box } from '@chakra-ui/react';

export default const StyledBox = () => {
	return (
		<Box
			as="button"
			bg="tomato"
			color="white"
			p={4}
			borderRadius="md"
			_hover={{ bg: "red.500" }}
		>
			Click Me
		</Box>
	);
};

7. TypeScript Support:

import React, { FC } from 'react';
import { Button } from '@chakra-ui/react';

interface MyButtonProps {
	label: string;
}

const MyButton: FC<MyButtonProps> = ({ label }) => {
	return (
		<Button colorScheme="green">{label}</Button>
	);
};
